# ── Step 1: Import library ──────────────────────────────
import pandas as pd
import numpy  as np

# ── Step 2: กำหนด path ไฟล์ ──────────────────────────────
PATH_CUSTOMERS = 'data/customers.csv'
PATH_PRODUCTS  = 'data/products.csv'
PATH_SALES     = 'data/sales_transactions.csv'

# ── Step 3: โหลดไฟล์ CSV ──────────────────────────────────
customers = pd.read_csv(PATH_CUSTOMERS)
products  = pd.read_csv(PATH_PRODUCTS)
sales     = pd.read_csv(PATH_SALES)

print("โหลดข้อมูลสำเร็จ ✓")

ประเภท dtype ความหมาย ตัวอย่าง
int64 จำนวนเต็ม ไม่มีทศนิยม customer_id, quantity
float64 ทศนิยม หรือตัวเลขที่มีค่า NaN age, price, total_spending
object / str ข้อความ (คอลัมน์ที่ไม่ใช่ตัวเลข) first_name, gender, date
bool ค่า True / False churned, active
datetime64 วันที่-เวลา (หลัง convert แล้ว) date, join_date

📅 Invalid Dates — วันที่ไม่ถูกต้อง

ลองแปลงวันที่ด้วย pd.to_datetime() แล้วนับแถวที่แปลงไม่ได้ (เป็น NaT)

task1_1_exploration.py 
# ตรวจสอบวันที่ที่ไม่ถูกต้อง
# errors='coerce' → แปลงค่าที่ผิดรูปแบบเป็น NaT (Not a Time)

before = customers['join_date'].isnull().sum()
parsed = pd.to_datetime(customers['join_date'], errors='coerce')
after  = parsed.isnull().sum()
invalid_dates = after - before  # แถวใหม่ที่กลายเป็น NaT หลังแปลง

print(f"Invalid Dates (join_date): {invalid_dates} rows")

Invalid Dates (join_date): 3 rows

🔗 Invalid IDs — รหัสที่ไม่มีอยู่ในไฟล์อ้างอิง

เช็คว่า product_id และ customer_id ใน sales มีอยู่จริงในไฟล์ products/customers ด้วย .isin()

task1_1_exploration.py 
# รวบรวม ID ที่ถูกต้องจากไฟล์อ้างอิง
valid_product_ids  = set(products['product_id'].dropna().astype(int))
valid_customer_ids = set(customers['customer_id'])

# นับ ID ใน sales ที่ไม่มีในไฟล์อ้างอิง
invalid_pid = (~sales['product_id'].isin(valid_product_ids)).sum()
invalid_cid = (~sales['customer_id'].isin(valid_customer_ids)).sum()

print(f"Invalid product_id  (not in products.csv):  {invalid_pid} rows")
print(f"Invalid customer_id (not in customers.csv): {invalid_cid} rows")
ผลลัพธ์จากข้อมูลจริง
Invalid product_id (not in products.csv): 3 rows Invalid customer_id (not in customers.csv): 0 rows
🚨 Unexpected Values — ค่าที่ไม่ตรงตาม Data Dictionary

ตรวจสอบว่าค่าในคอลัมน์ประเภทหมวดหมู่ตรงกับที่กำหนดใน Data Dictionary ด้วย .isin()

task1_1_exploration.py 
# ── gender ควรมีแค่ 'M' หรือ 'F' ────────────────────────
expected_gender = {'M', 'F'}
bad_gender = customers[
    customers['gender'].notna() &
    ~customers['gender'].isin(expected_gender)
]
print(f"Unexpected gender: {len(bad_gender)} rows → {customers['gender'].unique().tolist()}")

# ── membership_status ควรเป็น Basic/Silver/Gold ──────────
expected_mem = {'Basic', 'Silver', 'Gold'}
bad_mem = (~customers['membership_status'].isin(expected_mem)).sum()
print(f"Unexpected membership_status: {bad_mem} rows")

# ── category ควรเป็น Pastries/Bread/Tarte ────────────────
expected_cat = {'Pastries', 'Bread', 'Tarte'}
bad_cat = products[
    products['category'].notna() &
    ~products['category'].str.strip().isin(expected_cat)
]
print(f"Unexpected category: {len(bad_cat)} rows → {products['category'].dropna().unique().tolist()}")

# ── churned ควรเป็น True/False เท่านั้น ─────────────────
expected_churn = {'True', 'False'}
bad_churn = (~customers['churned'].isin(expected_churn)).sum()
print(f"Unexpected churned: {bad_churn} rows → {customers['churned'].unique().tolist()}")

🔧 Formatting Issues — ช่องว่างหน้า/หลังข้อความ

เปรียบเทียบข้อความก่อนและหลัง .strip() ถ้าต่างกัน แปลว่ามี extra spaces

task1_1_exploration.py 
# ฟังก์ชันตรวจช่องว่างหน้า/หลัง
def count_extra_spaces(df, cols):
    total = 0
    for col in cols:
        if col in df.columns:
            mask = (
                df[col].dropna().astype(str).str.strip()
                !=
                df[col].dropna().astype(str)
            )
            total += mask.sum()
    return total

# ตรวจแต่ละไฟล์
fmt_customers = count_extra_spaces(
    customers, ['first_name','last_name','gender','membership_status','preferred_category']
)
fmt_products = count_extra_spaces(
    products, ['product_name','category','ingredients']
)
fmt_sales = count_extra_spaces(
    sales, ['payment_method','channel']
)

print(f"Formatting Issues (customers): {fmt_customers} rows")
print(f"Formatting Issues (products):  {fmt_products} rows")
print(f"Formatting Issues (sales):     {fmt_sales} rows")
