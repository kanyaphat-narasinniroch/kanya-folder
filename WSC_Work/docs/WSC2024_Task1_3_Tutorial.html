def make_line_chart(x, y, title, ylabel, color, fname):
    """วาด Line Chart และบันทึกเป็น PNG"""

    # ── สร้างกระดาษวาด (Figure) ขนาด 9×3.5 นิ้ว ──
    fig, ax = plt.subplots(figsize=(9, 3.5), facecolor='#F8FAFC')
    ax.set_facecolor('#F8FAFC')

    # ── วาดเส้นกราฟ ────────────────────────────────
    ax.plot(
        x, y,
        color=color,       # สีเส้น
        linewidth=2.5,      # ความหนาเส้น
        marker='o',         # จุดที่แต่ละ data point
        markersize=5,      # ขนาดจุด
        zorder=3           # วาดทับเส้นอื่น
    )

    # ── เติมพื้นที่ใต้เส้น (ทำให้ดูดีขึ้น) ────────
    ax.fill_between(x, y, alpha=0.1, color=color)

    # ── หัวข้อและป้ายกำกับ ────────────────────────
    ax.set_title(title,  fontsize=13, fontweight='bold', pad=12)
    ax.set_ylabel(ylabel, fontsize=10)
    ax.set_xlabel('Month', fontsize=10)

    # ── หมุนป้ายแกน X ──────────────────────────────
    ax.tick_params(axis='x', rotation=30, labelsize=9)

    # ── จัดรูปแบบตัวเลขแกน Y เป็น 1,000 ──────────
    ax.yaxis.set_major_formatter(
        mticker.FuncFormatter(lambda v, _: f'{v:,.0f}')
    )

    # ── เส้นตารางแนวนอน ─────────────────────────
    ax.grid(axis='y', linestyle='--', alpha=0.4)

    # ── ปรับ layout อัตโนมัติ ──────────────────
    plt.tight_layout()

    # ── บันทึกเป็น PNG ──────────────────────────
    path = f'output/{fname}'
    plt.savefig(path, dpi=150, bbox_inches='tight')
    plt.close()   # คืน memory — สำคัญมากเมื่อสร้างหลายกราฟ
    return path   # ส่ง path กลับไปใช้ตอนสร้าง PDF
📌 parameter ของฟังก์ชัน
parameterความหมายตัวอย่าง
xค่าแกน X (ชื่อเดือน)monthly['month_str']
yค่าแกน Y (ตัวชี้วัด)monthly['total_revenue']
titleหัวข้อกราฟ'Total Sales Revenue...'
ylabelป้ายกำกับแกน Y'Revenue (€)'
colorสีเส้นกราฟ'#3B82F6'
fnameชื่อไฟล์ PNG'chart_revenue.png'

from reportlab.lib.pagesizes import A4
from reportlab.platypus import (
    SimpleDocTemplate, Image, Spacer,
    Table, TableStyle, Paragraph
)
from reportlab.lib.styles import ParagraphStyle
from reportlab.lib import colors
from reportlab.lib.units import cm

# ── กำหนด Styles ─────────────────────────────────────────
title_style = ParagraphStyle('t', fontSize=18, fontName='Helvetica-Bold',
    spaceAfter=4)
head_style  = ParagraphStyle('h', fontSize=13, fontName='Helvetica-Bold',
    spaceBefore=12, spaceAfter=6)

# ── สร้าง Document ───────────────────────────────────────
doc = SimpleDocTemplate(
    'output/Session1_SalesTrends.pdf',
    pagesize=A4,
    rightMargin=2*cm, leftMargin=2*cm,
    topMargin=2*cm,   bottomMargin=2*cm
)

# ── สร้าง story list ─────────────────────────────────────
story = []

# หัวข้อรายงาน
story.append(Paragraph("Session 1 — Sales Trend Analysis", title_style))

# ใส่กราฟทั้ง 3 พร้อมหัวข้อ
for chart, heading in [
    (chart1, "Total Sales Revenue per Month"),
    (chart2, "Number of Transactions per Month"),
    (chart3, "Average Order Value per Month"),
]:
    story.append(Paragraph(heading, head_style))
    story.append(Image(chart, width=16*cm, height=6*cm))
    story.append(Spacer(1, 8))          # ช่องว่าง 8pt

# ── ตาราง Top 3 ──────────────────────────────────────────
story.append(Paragraph("Top 3 Months by Sales Revenue", head_style))

# สร้างข้อมูลตาราง: แถวแรก = header
tbl_data = [['Month', 'Total Revenue (€)']]
for _, row in top3.iterrows():
    tbl_data.append([
        row['month_str'],
        f"€ {row['total_revenue']:,.2f}"   # จัดรูปแบบตัวเลข
    ])

# สร้างตาราง
tbl = Table(tbl_data, colWidths=[8*cm, 8*cm])
tbl.setStyle(TableStyle([
    ('BACKGROUND', (0,0), (-1,0), colors.HexColor('#1E3A5F')), # header bg
    ('TEXTCOLOR',  (0,0), (-1,0), colors.white),              # header text
    ('FONTNAME',   (0,0), (-1,0), 'Helvetica-Bold'),
    ('FONTSIZE',   (0,0), (-1,-1), 11),
    ('ALIGN',      (1,0), (1,-1), 'RIGHT'),               # จัดขวาคอลัมน์ 2
    ('ROWBACKGROUNDS', (0,1), (-1,-1),                       # สลับสีแถว
     [colors.HexColor('#F8FAFC'), colors.white]),
    ('GRID',       (0,0), (-1,-1), 0.5, colors.HexColor('#CBD5E1')),
    ('TOPPADDING',    (0,0), (-1,-1), 8),
    ('BOTTOMPADDING', (0,0), (-1,-1), 8),
]))
story.append(tbl)

# ── สร้าง PDF ─────────────────────────────────────────────
doc.build(story)
print("✓ บันทึก Session1_SalesTrends.pdf สำเร็จ")
reportlab classใช้ทำอะไร

Tableตารางข้อมูล กำหนดความกว้างแต่ละคอลัมน์

sales = pd.read_csv('data/sales_transactions.csv')
sales['date']          = pd.to_datetime(sales['date'], errors='coerce')

def make_line_chart(x, y, title, ylabel, color, fname):
    fig, ax = plt.subplots(figsize=(9, 3.5), facecolor='#F8FAFC')
    ax.set_facecolor('#F8FAFC')
    ax.plot(x, y, color=color, linewidth=2.5, marker='o', markersize=5, zorder=3)
    ax.fill_between(x, y, alpha=0.1, color=color)
    ax.set_title(title, fontsize=13, fontweight='bold', pad=12)
    ax.set_ylabel(ylabel, fontsize=10)
    ax.set_xlabel('Month', fontsize=10)
    ax.tick_params(axis='x', rotation=30, labelsize=9)
    ax.yaxis.set_major_formatter(mticker.FuncFormatter(lambda v,_: f'{v:,.0f}'))
    for spine in ax.spines.values(): spine.set_edgecolor('#E2E8F0')
    ax.grid(axis='y', linestyle='--', alpha=0.4)
    plt.tight_layout()
    path = f'output/{fname}'
    plt.savefig(path, dpi=150, bbox_inches='tight')
    plt.close()
    return path
